auto increment no oracle - GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY
chave estrangeira - CONSTRAINT fk_id_pessoa FOREIGN KEY (id_pessoa) REFERENCES pessoas(id)

create table PESSOAS(
    id number GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    idFirebase varchar(255),
    nome varchar(255),
    sobrenome varchar(255),
    genero number,
    dataNascimento date,
    cpf number,
    graduacao number,
    especialidade number,
    fone number,
    email varchar(255)
);

create table ENDERECOS(
    id number GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    idPessoa number not null,
    cep number,
    uf varchar(2),
    municipio varchar(255),
    bairro varchar(255),
    rua varchar(255),
    numero number,
    complemento varchar(255),
    CONSTRAINT fk_idPessoa FOREIGN KEY (idPessoa) REFERENCES PESSOAS(id)
);
